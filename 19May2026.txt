
SELECT FIRST_NAME, SALARY FROM Worker WHERE SALARY BETWEEN 100000 AND 300000;
SELECT FIRST_NAME, SALARY FROM Worker WHERE SALARY NOT BETWEEN 100000 AND 300000;

SELECT FIRST_NAME, DEPARTMENT FROM Worker WHERE DEPARTMENT IN ('HR', 'Account');
SELECT FIRST_NAME, DEPARTMENT FROM Worker WHERE DEPARTMENT NOT IN ('HR', 'Account');

SELECT FIRST_NAME, SALARY, DEPARTMENT 
FROM Worker 
WHERE (SALARY BETWEEN 100000 AND 300000) 
  AND DEPARTMENT IN ('HR', 'Account');

SELECT FIRST_NAME, SALARY, DEPARTMENT 
FROM Worker 
WHERE (SALARY BETWEEN 80000 AND 500000) 
  AND DEPARTMENT NOT IN ('HR', 'Admin');
  
SELECT COUNT(*) AS TOTAL_WORKERS FROM Worker;

SELECT SUM(SALARY) AS TOTAL_SALARY_EXPENSE FROM Worker;

SELECT AVG(SALARY) AS AVERAGE_SALARY FROM Worker;

SELECT MIN(SALARY) AS LOWEST_SALARY FROM Worker;

SELECT MAX(SALARY) AS HIGHEST_SALARY FROM Worker;

SELECT DEPARTMENT, MIN(SALARY) AS MIN_SALARY 
FROM Worker 
GROUP BY DEPARTMENT;

SELECT FIRST_NAME FROM Worker WHERE FIRST_NAME LIKE 'M%';

SELECT FIRST_NAME FROM Worker WHERE FIRST_NAME LIKE '%a';

SELECT FIRST_NAME FROM Worker WHERE FIRST_NAME LIKE '%i%';

SELECT FIRST_NAME FROM Worker WHERE FIRST_NAME LIKE '_i%';

SELECT FIRST_NAME, SALARY 
FROM Worker 
WHERE DEPARTMENT = 'Admin' 
  AND FIRST_NAME NOT LIKE '%a%'
  AND SALARY = (
      SELECT MIN(SALARY) 
      FROM Worker 
      WHERE DEPARTMENT = 'Admin' 
        AND FIRST_NAME NOT LIKE '%a%'
  );
  
  SELECT FIRST_NAME, SALARY,
    CASE 
        WHEN SALARY >= 300000 THEN 'High Salary'
        WHEN SALARY >= 100000 THEN 'Medium Salary'
        ELSE 'Low Salary'
    END AS SALARY_STATUS
FROM Worker;

SELECT FIRST_NAME, DEPARTMENT, SALARY,
    CASE 
        WHEN DEPARTMENT = 'Admin' AND SALARY > 100000 THEN 'Senior Admin'
        WHEN DEPARTMENT = 'HR' AND SALARY > 100000 THEN 'Senior HR'
        ELSE 'Associate'
    END AS ROLE_TIER
FROM Worker;

SELECT 
    COUNT(CASE WHEN DEPARTMENT = 'Admin' THEN 1 END) AS ADMIN_COUNT,
    COUNT(CASE WHEN DEPARTMENT = 'HR' THEN 1 END) AS HR_COUNT,
    COUNT(CASE WHEN DEPARTMENT = 'Account' THEN 1 END) AS ACCOUNT_COUNT
FROM Worker;


CREATE VIEW Worker_Public_Info AS
SELECT FIRST_NAME, LAST_NAME, DEPARTMENT
FROM Worker;
